\section{Introduction}
This project explores and optimize modular reductions for prime numbers  $p < 2^{32}$ using SIMD Vectorisation (AVX2, AVX512, NEON) along with various different techniques and algorithmes.

\textbf{Objective: } The aim of this study is to be able to realize an optimized implementation with minimal performance cost for modular reductions over 32 bits.

\section{Fundamental Cost of Modular Reduction}
The most naive way to compute $a \pmod{n}$ is to perform the operation :

\begin{equation}
    a - \left\lfloor \frac{a}{n} \right\rfloor \cdot n
\end{equation}

However, on modern processor architectures (Intel, AMD,\dots), integer division has significantly higher computational cost than other operations like additions and multiplications.

Since equation (1) requires a divsion and substraction operation, its considered highly inefficient from both a software and hardware perspective.

\section{Shoup's Modular Multiplication}
Shoup's algorithm is centraled around precomputation as a work around the costly division operations

\begin{algorithm}[H]
    \caption{Shoup's modular multiplication algorithm}

    % Input and Output definitions
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}

    % The content
    \Input{$A, B \in [0, p[$, we suppose that $p < \beta$/2 \\
    \hspace{1.2cm} precompute $B' = \lfloor B\beta/p \rfloor$}
    \Output{$R = AB \bmod p$}

    \BlankLine

    % Steps
    $Q \leftarrow \lfloor AB'/\beta \rfloor$ \quad ;\\
    $R \leftarrow (AB - Qp) \bmod \beta$ \quad ;

    \If{$R \ge p$}{
        $R \leftarrow R - p$\;
    }
\end{algorithm}



\subsection{Input Constraint:}
To ensure the correctness of the modular reduction and prevent register overflow during SIMD operations,
the following conditions must be satisfied:
\begin{itemize}
    \item \textbf{Word Size ($\beta$):} Set to $2^{32}$ for 32-bit modular arithmetic
    \item \textbf{Input 1 ($A$):} Integer defined over 32 bits
    \item \textbf{Input 2 ($B$):} Integers within the range $[0, p[$
    \item \textbf{Modulus ($p$):} A prime number defined over 31 bits ($p < 2^{31}$)
\end{itemize}

\subsection{Theorem}
Let $p$ be a prime number such that $p < \beta/2$ where $\beta = 2^{32}$. Given integers $A$ an integer defined over 32 bits and
$ B \in [0, p)$, let $B'$ be the precomputed value $B' = \lfloor B\beta/p \rfloor$.
If the estimated quotient is defined as $Q = \lfloor AB'/\beta \rfloor$, then the intermediate remainder $R = AB - Qp$ satisfies the following condition:
$$0 \le AB - Qp < 2p$$

\subsection{Input Hypothesis:}
\textbf{Can p be any integer over 32 bits?} \\
No, one particularly interesting case where Shoup's algorithme will fail is for $p \ge 2^{31}$.
Since we know that:
$$0 \le AB - Qp < 2p$$

if $p \ge 2^{31}$ thus $2p \ge 2^{32}$. Therefore, in some cases $AB - Qp$ may exceed $2^{32}$. This causes tremendous errors in the calculation because
all bits beyond 32 bits are destroyed when we mod $\beta$

\textbf{Does p have to neccesarily be a prime number?} \\
No, the algorithm mainly uses precomputations (Barrett Reduction), optimized for caculation of $a \mod n$ without constraints on n.
But for the sake of this project, n is a prime number (p).

\textbf{Can b be any integer over 32 bits?} \\
No, suppose $ b > p $, we can rewrite $b$ as:
\[b = q'\cdot p + r'\]
Thus the caculations for the precomputed value $B'$ is:
\[B' = \left\lfloor\frac{(q' \cdot p + f)\cdot 2^{32}}{p}\right\rfloor\]
\[B' = q' \cdot 2^{32} + \left\lfloor\frac{ f\cdot 2^{32}}{p}\right\rfloor\]
With a focus on 32 bits in this project, $B'$ in the case of $b>p$ cannot be stored within 32 bits and will overflow.\\
\textbf{Can a be any integer over 32 bits?} \\
Yes, a can be any integer defined over 32 bits
\subsection{Procomputational Logic (Barrett Reduction)}
The precomputed value $B'$ is defined over 32 bits as:
\[B'= \lfloor (B \cdot 2^{32}) / p \rfloor \]

When computing the $a \cdot b \pmod{p}$ :
\begin{equation*}
    A \cdot B \pmod{p} = A \cdot B -  \left\lfloor \frac{A \cdot B}{p} \right\rfloor \cdot p
\end{equation*}
We can rewrite this:
\begin{align*}
    A \cdot B \pmod{p} & = A \cdot B -  \left\lfloor \frac{A \cdot \left\lfloor (\frac{B \cdot 2^{32}}{p}) \right\rfloor}{2^{32}} \right\rfloor \cdot p \\
                       & \approx A \cdot B -  \left\lfloor \frac{A \cdot B'}{2^{32}} \right\rfloor \cdot p                                              \\
\end{align*}



\subsection{Fundamental Principal}
Given x the quotient of 2 32 bits integers: \\
The proof of this algorithm based on the non decimal value of ($\lfloor x \rfloor$):
$$x - 1 < \lfloor x \rfloor \le x$$
We can rewrite as:
$$0 \le x - \lfloor x \rfloor < 1$$


\subsection{Demonstration:}
The algorithm uses the precomputation $B' = \lfloor B\beta/p \rfloor$ such that:
\[0 \le \frac{B\beta}{p} - \left\lfloor \frac{B\beta}{p} \right\rfloor < 1 \]
\begin{equation}
    0 \le \frac{B\beta}{p} - B' < 1
\end{equation}

Using the same logic, the algorithm estimates the quotient $Q = \lfloor AB'/\beta \rfloor$  such that:
\[0 \le \frac{AB'}{\beta} - \left\lfloor\frac{AB'}{\beta}\right\rfloor < 1\]
\begin{equation}
    0 \le \frac{AB'}{\beta} - Q < 1
\end{equation}

We now prouve that remainder $R = (A \cdot B -Q \cdot p)$ is within the interval [0, 2p[:

We firstly multiply the equation (1) by $Ap/\beta$:
\[0 \le \left( \frac{B\beta}{p} - B' \right) \cdot \frac{Ap}{\beta} < 1 \cdot \frac{Ap}{\beta}\]
\begin{equation}
    0 \le AB - \frac{B'Tp}{\beta} < \frac{Ap}{\beta}
\end{equation}

Similarly, we multiply the equation (2) by $p$:
\[0 \le \left( \frac{AB'}{\beta} - Q \right) \cdot p < 1 \cdot p\]
\begin{equation}
    0 \le \frac{AB'p}{\beta} - Qp < p
\end{equation}

By adding the equation (3) and (4), we deduce:
\[0 \le (AB - \cancel{\frac{AB'p}{\beta}}) + (\cancel{\frac{AB'p}{\beta}} - Qp) < \frac{Ap}{\beta} + p\]
Simplifying this result, we are left with:
\begin{align*}
    0 \le AB - Qp & < p + \frac{Ap}{\beta}          \\
    0 \le AB - Qp & < p \cdot (1 + \frac{A}{\beta})
\end{align*}

By constraint, $A \in [0, p[$ and $p < \beta$ thus $A < \beta$. Therefore we can rewrite:
\begin{align*}
    0 \le AB - Qp & < p \cdot (1 + \frac{A}{\beta}) < p\cdot (1 + \frac{\beta}{\beta}) \\
    0 \le AB - Qp & < p \cdot (1 + 1)                                                  \\
    0 \le AB - Qp & < 2p                                                               \\
    0 \le R       & < 2p
\end{align*}
\textbf{Conclusion:}
We have now prouven that $R \in [0, 2p[$ the final if statement of the helps correct the remainder. If $R > p$ the
modular reduction needs 1 more step $R - p$, if not $R$ is the correct result

\subsection{Avantages}

Using the precomputed value B' to approxiamtely calculate $\left\lfloor \frac{A \cdot B}{p} \right\rfloor$ improves on the naive method by 2 ways:
\begin{enumerate}
    \item \textbf{Storing the Inverse of $B$:}
          $B'$ is computed with 1 division and can be stored to be used in other modular reduction provided the same B provided the same $p$
    \item \textbf{Shoup's algorithm replaces division operations (slow) with multiplications and additions (fast):}
          $\frac{A \cdot B'}{2^{32}}$ has only 1 multiplication operation and a bit shift to the right of 32 bits. On a CPU, both multiplication and bit shift operations can be done efficiently compared to division
\end{enumerate}
\subsection{SIMD Vectorization Implementation}
We decided to create 2 versions for implementating a high performance Shoup algorithm
\subsubsection{AVX}
For the AVX version, we decided to use 2 main AVX function:
\_mm256\_mul\_epu32 and \_mm256\_mul\_epi32. \\

\textbf{Intel \_mm256\_mul\_epu32}\\
This function treats the input as unsigned 32-bit integers. However, it only multiplies the even-indexed slots (0, 2, 4, 6) of the vector to make room for the full 64-bit results.
\begin{lstlisting} [caption = Auxilary Function of Shoup's Modular Reduction ]
    static inline __m256i _shoup_avx(__m256i va, 
                                     __m256i vb, 
                                     __m256i vb_precomp, 
                                     __m256i vp, 
                                     __m256i mask32)
{
    // 1. q = (a * b_precomp) >> 32:
    __m256i q = _mm256_mul_epu32(va, vb_precomp);
    q = _mm256_srli_epi64(q, 32);

    // 2. ab:
    __m256i ab = _mm256_mul_epu32(va, vb);

    // 3. qp:
    __m256i qp = _mm256_mul_epu32(q, vp);

    // 4. c = (ab - qp) & mask:
    __m256i c = _mm256_sub_epi64(ab, qp);
    c = _mm256_and_si256(c, mask32);

    // 5. if (c >= p) c -= p
    __m256i compare = _mm256_cmpgt_epi64(vp, c);
    __m256i sub_mask = _mm256_andnot_si256(compare, vp);

    return _mm256_sub_epi64(c, sub_mask);
}
\end{lstlisting}

\begin{lstlisting}  [caption = AVX2 Implementation of Shoup's Modular Reduction ]
    static Vector _shoup_scale_avx(Parameters param, Vector v)
{
    /*1. Preloading b, b_precomp, p and 32 bit mask*/
    int size = v.size;
    Vector res = init_vector(size);
    __m256i vb = _mm256_set1_epi64x(param.b);
    __m256i vb_precomp = _mm256_set1_epi64x(param.b_precomp);
    __m256i vp = _mm256_set1_epi64x(param.p);
    __m256i mask_32 = _mm256_set1_epi64x(0xFFFFFFFF);

    int i = 0;
    for (; i + 7 < size; i += 8)
    {
        /* 2. Load 8 elements [v7 v6 v5 v4 v3 v2 v1 v0]: */
        __m256i va_input = _mm256_loadu_si256((__m256i const *)
                            (v.elements));
        /* 3. Split into two sets of 4 : va_input and va_odd */
        // Elements [null, 7, null, 5, null , 3, null, 1]
        __m256i va_odd = _mm256_srli_epi64(va_input, 32); 
        
        /* 4. Process Even Lanes: */
        __m256i c_even = _shoup_avx(va_input, vb, 
                                      vb_precomp, vp, mask_32);
       
        /* 5. Process Odd Lanes: */
        __m256i c_odd = _shoup_avx(va_odd, vb, vb_precomp,
                                   vp, mask_32);
        
        /* 6. Shift odd by 32 bits to the left results back 
        and merge: */
        // [7, null, 5, null, 3, null, 1, null]
        __m256i res_odd_shifted = _mm256_slli_epi64(c_odd, 32); 
        
        // [7, null, 5, null, 3, null, 1, null] 
        //or null, 6, null, 4, null, 2, null, 0]
        __m256i res_combined = _mm256_or_si256(c_even, 
                                               res_odd_shifted); 
        /* 7. Store: */
        _mm256_storeu_si256((__m256i *)(res.elements), 
                            res_combined);
    }
    for (; i < size; i++)
        res.elements[i] = shoup(v.elements[i], param.b, 
                                param.b_precomp, param.p);
    return res;
}
\end{lstlisting}

\textbf{Intel \_mm256\_mullo\_epi32}\\
This function treats the input as signed 32-bit integers. It multiplies all 8 pairs of numbers but only stores the bottom half of the result.
\begin{lstlisting} [caption = Auxilary Function of Shoup’s Modular Reduction (Version mullo)]
    static inline __m256i _shoup_avx_mullo(__m256i va, 
                                           __m256i vb, 
                                           __m256i vb_precomp, 
                                           __m256i vp)
{
    // 1. q = (a * b_precomp) >> 32:
    __m256i q = _mm256_mul_epu32(va, vb_precomp);
    q = _mm256_srli_epi64(q, 32);

    // 2. ab:
    __m256i ab = _mm256_mullo_epi32(va, vb);

    // 3. qp
    __m256i qp = _mm256_mullo_epi32(q, vp);

    // 4. c = ab - qp
    __m256i c = _mm256_sub_epi32(ab, qp);

    // 5. if (c >= p) c -= p
    __m256i compare = _mm256_cmpgt_epi64(vp, c);
    __m256i sub_mask = _mm256_andnot_si256(compare, vp);

    return _mm256_sub_epi64(c, sub_mask);
}
\end{lstlisting}

\textbf{Intel \_mm256\_mullo\_epi32 version 2:}\\
Derived from the previous version, this version aims to regroup the vector registers immediately after the
multiplication step 
\[q = \lfloor (a \cdot b_{precomp}) / 2^{32} \rfloor\]
By restoring the standard 8-lane 32-bit layout early, the function eliminates the need to perform parrallel logic paths.
This in theory should reduce the work load.

In order to implement this version, we create an auxilary function \textbf{\_fake\_mulhi} which emulates a native 'multiply-high' 
instruction for unsigned 32-bit integers. It computes the full 64-bit products of two packed vectors, 
extracts the most significant 32 bits from each result, and repacks them into a single 256-bit 
destination register.

\begin{lstlisting} [caption = Auxilary Function \_fake\_mulhi]
    static inline __m256i _fake_mulhi(__m256i va, __m256i vb)
{
    /* 1. Multiply even lanes (0, 2, 4, 6) */
    __m256i res_even = _mm256_mul_epu32(va, vb);

    /* 1. Multiply odd lanes (1, 3, 5, 7) */
    __m256i res_odd = _mm256_mul_epu32(_mm256_srli_epi64(va, 32), vb);

    /* 3. Extract high 32 bits of even-lane products*/
    __m256i hi_even = _mm256_srli_epi64(res_even, 32);

    /* 4. Extract high 32 bits of odd-lane products */
    __m256i mask = _mm256_set1_epi64x(0xFFFFFFFF00000000ULL);
    __m256i hi_odd = _mm256_and_si256(res_odd, mask);

    /*5. Regroup*/
    return _mm256_or_si256(hi_even, hi_odd);
}
\end{lstlisting}
\begin{lstlisting} [caption = Auxilary Function of Shoup’s Modular Reduction (Version 2 mullo)]
    static inline __m256i _shoup_mullo_avx2_v2(__m256i va, 
                                               __m256i vb, 
                                               __m256i vb_precomp, 
                                               __m256i vp)
{
    // 1. q = (a * b_precomp) >> 32:
    __m256i vq = _fake_mulhi(va, vb_precomp);

    // 2. ab:
    __m256i vab = _mm256_mullo_epi32(va, vb);

    // 3. qp:
    __m256i vqp = _mm256_mullo_epi32(vq, vp);

    // 4. c = (ab - qp):
    __m256i vc = _mm256_sub_epi32(vab, vqp);

    //Unsigned Comparison using min
    // 5. if (c >= p) c -= p
    __m256i v_min = _mm256_min_epu32(vc, vp);
    __m256i is_ge = _mm256_cmpeq_epi32(v_min, vp);
    __m256i sub_mask = _mm256_and_si256(is_ge, vp);
    return _mm256_sub_epi64(vc, sub_mask);
}
\end{lstlisting}

\begin{lstlisting} [caption = AVX2 Implementation of Shoup’s Modular Reduction (Version 2 mullo)]
    Vector shoup_scale_mullo_avx2_v2(Parameters param, Vector v)
{
    /* 1. Preloading b , b_precomp , p */
    ulong size = v.size;
    Vector res = init_vector(size);
    __m256i vb = _mm256_set1_epi32(param.b);
    __m256i vb_precomp = _mm256_set1_epi32(param.b_precomp);
    __m256i vp = _mm256_set1_epi32(param.p);
    ulong i = 0;
    for (; i + 7 < size; i += 8)
    {
        /* 2. Load 8 elements [ v7 v6 v5 v4 v3 v2 v1 v0 ]: */
        __m256i va = _mm256_loadu_si256((__m256i const *)(v.elements + i));

        /* 3. Apply Shoup */
        __m256i vc = _shoup_mullo_avx2_v2(va, vb, vb_precomp, vp);

        /* 4. Store : */
        _mm256_storeu_si256((__m256i *)(res.elements + i), vc);
    }

    /*5. Apply Shoup to the remaning elements*/
    for (; i < size; i++)
        *(res.elements + i) = shoup(*(v.elements + i), param.b, param.b_precomp, param.p);
    return res;
}
